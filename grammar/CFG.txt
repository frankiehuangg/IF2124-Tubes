# Line
S -> LINE LINE ENDLINE | LINE ENDLINE
LINE -> LET | CONST | VALUE | OBJECT | ARRAY | OBJECT | ARRAY | CONDITIONAL | SWITCH | FOR | WHILE | DOWHILE | FUNCTION | CATCHBLOCK



# Operator
ARITHMETIC -> PLUS | MINUS | MULT | POWER | DIVIDE | MODULO
INCDEC -> INCREMENT | DECREMENT

ASSIGNMENT -> EQUAL | PLUSEQUAL | MINEQUAL | MULTEQUAL | DIVEQUAL | MODEQUAL | POWEQUAL | SHIFTLEFTEQUAL | SHIFTARITHRIGHTEQUAL | SHIFTLOGICRIGHTEQUAL | ANDEQUAL | XOREQUAL | OREQUAL | LOGICANDEQUAL | LOGICOREQUAL

COMPARISON -> COMPARE | STRICTCOMPARE | NOTCOMPARE | STRICTNOTCOMPARE | BIGGERTHAN | SMALLERTHAN | BIGGEREQUALTHAN | SMALLEREQUALTHAN

LOGICAL -> LOGICAND | LOGICOR

BITWISE -> AND | OR | XOR | SHIFTLEFT | SHIFTARITHRIGHT | SHIFTLOGICRIGHT



# Primitive Data Types
INTEGER -> NUMBER | NUMBER DOT NUMBER

BOOLEAN -> OPENPAR BOOLEAN CLOSEPAR | BOOLEAN LOGICAL BOOLEAN | LOGICNOT BOOLEAN | VALUE COMPARISON VALUE | TRUE | FALSE

VALUE -> OPENPAR VALUE CLOSEPAR | INTEGER | STRING | BOOLEAN | NULL | VALUE COMPARISON VALUE | VALUE ARITHMETIC VALUE | VALUE INCDEC | VALUE BITWISE VALUE | NOT VALUE | NULL | UNDEFINED



# Statements 
LET -> ASSIGN_LET TEXT ASSIGNMENT VALUE | ASSIGN_VAR TEXT ASSIGNMENT VALUE
CONST -> ASSIGN_CONST TEXT ASSIGNMENT VALUE



# Object 
OBJECT_PROP -> OBJECT_PROP COMMA OBJECT_PROP | TEXT COLON VALUE | TEXT COLON FUNCTION | TEXT
OBJECT -> ASSIGN_CONST TEXT EQUAL OPENBRAC OBJECT_PROP CLOSEBRAC
OBJECT_EXP -> OBJECT_EXP ENDLINE OBJECT_EXP | DELETE TEXT DOT TEXT ENDLINE



# Array 
ARRAY_PROP -> ARRAY_PROP COMMA ARRAY_PROP | VALUE
ARRAY -> ASSIGN_CONST TEXT EQUAL ARRAY_OPEN ARRAY_PROP ARRAY_CLOSE



# Conditional 
IF -> COND_IF OPENPAR BOOLEAN CLOSEPAR OPENBRAC S CLOSEPAR | COND_IF OPENPAR BOOLEAN CLOSEPAR LINE ENDLINE
ELIF -> ELIF ELIF | COND_ELSE COND_IF OPENPAR BOOLEAN CLOSEPAR OPENBRAC S CLOSEBRAC | COND_ELSE COND_IF OPENBRAC BOOLEAN CLOSEBRAC LINE ENDLINE
ELSE -> COND_ELSE OPENBRAC S CLOSEBRAC | COND_ELSE LINE ENDLINE

CONDITIONAL -> IF ELIF ELSE | IF ELSE | IF 



# Switch 
BREAK -> SWITCH_BREAK ENDLINE
CASE -> CASE CASE | SWITCH_CASE TEXT COLON S BREAK | SWITCH_CASE TEXT COLON S
DEFAULT -> SWITCH_DEFAULT COLON S BREAK | SWITCH_DEFAULT COLON S
SWITCH -> SWITCH_SWITCH OPENPAR TEXT CLOSEPAR OPENBRAC CASE DEFAULT CLOSEBRAC | SWITCH_SWITCH OPENPAR TEXT CLOSEPAR OPENBRAC CASE CLOSEBRAC



# Loops 
LOOP_PROP -> LOOP_PROP LOOP_PROP | S | BREAK ENDLINE | CONTINUE ENDLINE

EXP_ONE -> EXP_ONE COMMA EXP_ONE | LET
EXP_THR -> EXP_THR COMMA EXP_THR | TEXT INCDEC
EXPRESSION -> EXP_ONE SEMICOLON BOOLEAN SEMICOLON EXP_THR | SEMICOLON BOOLEAN SEMICOLON EXP_THR | EXP_ONE SEMICOLON SEMICOLON EXP_THR | EXP_ONE SEMICOLON BOOLEAN SEMICOLON | SEMICOLON SEMICOLON EXP_THR | SEMICOLON BOOLEAN SEMICOLON | EXP_ONE SEMICOLON SEMICOLON | SEMICOLON SEMICOLON SEMICOLON

FOR -> LOOP_FOR OPENPAR EXPRESSION CLOSEPAR OPENBRAC LOOP_PROP CLOSEBRAC  | LOOP_FOR OPENPAR ASSIGN_LET TEXT LOOP_IN TEXT CLOSEPAR OPENBRAC LOOP_PROP CLOSEBRAC | LOOP_FOR OPENPAR ASSIGN_LET TEXT of TEXT CLOSEPAR OPENBRAC LOOP_PROP CLOSEBRAC


WHILE -> LOOP_WHILE OPENPAR BOOLEAN CLOSEPAR OPENBRAC LOOP_PROP CLOSEBRAC
DOWHILE -> LOOP_DO OPENBRAC LOOP_PROP CLOSEBRAC LOOP_WHILE OPENPAR BOOLEAN CLOSEPAR ENDLINE


# Functions 
PARAMETER -> PARAMETER COMMA PARAMETER | TEXT
FUNC_PROP -> FUNC_PROP FUNC_PROP | S | FUNC_RETURN VALUE

FUNCTION -> FUNC_FUNCTION TEXT OPENPAR PARAMETER CLOSEPAR OPENBRAC FUNC_PROP CLOSEBRAC | ASSIGN_CONST TEXT EQUAL FUNC_FUNCTION OPENPAR PARAMETER CLOSEPAR OPENBRAC FUNC_PROP CLOSEBRAC | OPENPAR PARAMETER CLOSEPAR FUNC_LINE OPENBRAC FUNC_PROP CLOSEBRAC



# Try Catch Finally
CATCH_PROP -> CATCH_PROP CATCH_PROP | S | CATCH_THROW STRING ENDLINE
TRY -> CATCH_TRY OPENBRAC CATCH_PROP CLOSEBRAC
CATCH -> CATCH_CATCH OPENPAR TEXT CLOSEPAR OPENBRAC CATCH_PROP CLOSEBRAC
FINALLY -> CATCH_FINALLY OPENBRAC CATCH_PROP CLOSEBRAC

CATCHBLOCK -> TRY CATCH FINALLY | TRY CATCH | TRY FINALLY
